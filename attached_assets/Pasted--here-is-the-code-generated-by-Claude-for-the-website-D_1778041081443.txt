<script>
  // ── ANIMATED HIGHWAY CANVAS ──
  (function(){
    const canvas = document.getElementById('highway-canvas');
    const ctx = canvas.getContext('2d');

    function resize(){
      canvas.width  = canvas.offsetWidth;
      canvas.height = canvas.offsetHeight;
    }
    resize();
    window.addEventListener('resize', resize);

    // ── Scene config ──
    const VP   = { x: 0.62, y: 0.42 };   // vanishing point (fractional)
    const ROAD_WIDTH_FAR  = 0.08;          // fraction of width at horizon
    const ROAD_WIDTH_NEAR = 1.1;           // fraction of width at bottom

    // Lane dash state
    const NUM_LANES = 4;
    let dashOffset = 0;
    const DASH_SPEED = 18;                 // px per frame at full stretch — feels like ~80 mph

    // Oncoming headlights
    const NUM_CARS = 6;
    const cars = Array.from({length: NUM_CARS}, (_, i) => ({
      t: i / NUM_CARS,                     // 0 = far, 1 = near
      lane: -(Math.random() * 0.3 + 0.1), // negative = oncoming side
      speed: 0.004 + Math.random() * 0.003,
      brightness: 0.6 + Math.random() * 0.4
    }));

    // Roadside lights / city glow orbs
    const NUM_LIGHTS = 22;
    const lights = Array.from({length: NUM_LIGHTS}, (_, i) => ({
      side: i % 2 === 0 ? 1 : -1,
      pos:  i / NUM_LIGHTS,
      speed: 0.003 + Math.random() * 0.002,
      color: Math.random() > 0.5 ? '#D4A843' : '#ffffff',
      size: 1.5 + Math.random() * 2.5
    }));

    // Stars (fixed)
    const NUM_STARS = 160;
    const stars = Array.from({length: NUM_STARS}, () => ({
      x: Math.random(),
      y: Math.random() * 0.42,             // only above horizon
      r: Math.random() * 1.2,
      a: 0.15 + Math.random() * 0.5
    }));

    // Perspective helpers
    function lerp(a, b, t){ return a + (b - a) * t; }

    function perspPoint(laneOffset, depth){
      // depth: 0=horizon, 1=bottom of screen
      const W = canvas.width, H = canvas.height;
      const vpX = VP.x * W, vpY = VP.y * H;
      const roadHalfFar  = ROAD_WIDTH_FAR  * W * 0.5;
      const roadHalfNear = ROAD_WIDTH_NEAR * W * 0.5;
      const roadHalf = lerp(roadHalfFar, roadHalfNear, depth);
      const x = vpX + laneOffset * roadHalf;
      const y = lerp(vpY, H * 1.05, depth);
      return { x, y };
    }

    function draw(){
      const W = canvas.width, H = canvas.height;
      const vpX = VP.x * W, vpY = VP.y * H;

      // ── Sky gradient ──
      const sky = ctx.createLinearGradient(0, 0, 0, vpY);
      sky.addColorStop(0,   '#010108');
      sky.addColorStop(0.5, '#03030F');
      sky.addColorStop(1,   '#0A0818');
      ctx.fillStyle = sky;
      ctx.fillRect(0, 0, W, vpY + 4);

      // ── Distant city glow on horizon ──
      const cityGlow = ctx.createRadialGradient(vpX, vpY, 0, vpX, vpY, W * 0.55);
      cityGlow.addColorStop(0,   'rgba(212,168,67,0.18)');
      cityGlow.addColorStop(0.3, 'rgba(255,140,50,0.06)');
      cityGlow.addColorStop(1,   'transparent');
      ctx.fillStyle = cityGlow;
      ctx.fillRect(0, 0, W, vpY + 80);

      // ── Stars ──
      stars.forEach(s => {
        ctx.beginPath();
        ctx.arc(s.x * W, s.y * H, s.r, 0, Math.PI * 2);
        ctx.fillStyle = `rgba(255,255,255,${s.a})`;
        ctx.fill();
      });

      // ── Road surface ──
      ctx.beginPath();
      const farL  = perspPoint(-1, 0);
      const farR  = perspPoint( 1, 0);
      const nearL = perspPoint(-1, 1);
      const nearR = perspPoint( 1, 1);
      ctx.moveTo(farL.x,  farL.y);
      ctx.lineTo(farR.x,  farR.y);
      ctx.lineTo(nearR.x, nearR.y);
      ctx.lineTo(nearL.x, nearL.y);
      ctx.closePath();
      const roadGrad = ctx.createLinearGradient(0, vpY, 0, H);
      roadGrad.addColorStop(0,   '#0E0E1A');
      roadGrad.addColorStop(0.4, '#111118');
      roadGrad.addColorStop(1,   '#0A0A14');
      ctx.fillStyle = roadGrad;
      ctx.fill();

      // ── Road edges (white lines) ──
      [[-1, 0.9], [1, 0.9]].forEach(([side, bright]) => {
        ctx.beginPath();
        ctx.moveTo(perspPoint(side, 0).x, perspPoint(side, 0).y);
        ctx.lineTo(perspPoint(side, 1).x, perspPoint(side, 1).y);
        ctx.strokeStyle = `rgba(255,255,255,${bright})`;
        ctx.lineWidth = lerp(1, 4, 0.5);
        ctx.stroke();
      });

      // ── Centre divider (double yellow) ──
      [-0.03, 0.03].forEach(off => {
        ctx.beginPath();
        ctx.moveTo(perspPoint(off, 0).x, perspPoint(off, 0).y);
        ctx.lineTo(perspPoint(off, 1).x, perspPoint(off, 1).y);
        ctx.strokeStyle = 'rgba(212,168,67,0.85)';
        ctx.lineWidth = 1.5;
        ctx.stroke();
      });

      // ── Dashed lane lines ──
      const lanePositions = [-0.5, 0.5]; // two interior lanes each side
      lanePositions.forEach(laneX => {
        const STEPS = 80;
        for(let i = 0; i < STEPS; i++){
          const t0 = i / STEPS;
          const t1 = (i + 0.5) / STEPS;
          // animated: shift t by dashOffset
          const tOff = dashOffset / STEPS;
          const ta = (t0 + tOff) % 1;
          const tb = (t1 + tOff) % 1;
          if(ta > tb) continue; // skip wrap-around glitch
          const p0 = perspPoint(laneX, ta);
          const p1 = perspPoint(laneX, tb);
          const alpha = lerp(0.1, 0.7, ta);
          const lw    = lerp(0.5, 3, ta);
          ctx.beginPath();
          ctx.moveTo(p0.x, p0.y);
          ctx.lineTo(p1.x, p1.y);
          ctx.strokeStyle = `rgba(255,255,255,${alpha})`;
          ctx.lineWidth = lw;
          ctx.stroke();
        }
      });

      // ── Oncoming headlights ──
      cars.forEach(car => {
        const p = perspPoint(car.lane, car.t);
        const baseSize = lerp(1, 18, car.t);

        // Headlight beams (cones)
        if(car.t > 0.1){
          [-1, 1].forEach(side => {
            const beamGrad = ctx.createRadialGradient(
              p.x + side * baseSize * 0.6, p.y, 0,
              p.x + side * baseSize * 0.6, p.y + baseSize * 3, baseSize * 4
            );
            beamGrad.addColorStop(0, `rgba(255,250,220,${0.25 * car.brightness * car.t})`);
            beamGrad.addColorStop(1, 'transparent');
            ctx.fillStyle = beamGrad;
            ctx.fillRect(0, vpY, W, H - vpY);
          });
        }

        // Headlight circles
        [-0.7, 0.7].forEach(side => {
          const hx = p.x + side * baseSize * 0.5;
          const hy = p.y;
          const g = ctx.createRadialGradient(hx, hy, 0, hx, hy, baseSize * 0.9);
          g.addColorStop(0,   `rgba(255,252,230,${car.brightness})`);
          g.addColorStop(0.4, `rgba(220,200,150,${car.brightness * 0.5})`);
          g.addColorStop(1,   'transparent');
          ctx.fillStyle = g;
          ctx.beginPath();
          ctx.arc(hx, hy, baseSize * 0.9, 0, Math.PI * 2);
          ctx.fill();
        });

        // Advance car toward viewer
        car.t += car.speed;
        if(car.t > 1.1) car.t = 0;
      });

      // ── Roadside pole lights ──
      lights.forEach(light => {
        const lx = light.side === 1 ? perspPoint(1.15, light.pos).x : perspPoint(-1.15, light.pos).x;
        const ly = perspPoint(0, light.pos).y;
        const sz = lerp(0.5, light.size, light.pos) * (light.pos < 0.05 ? light.pos * 20 : 1);
        if(light.pos > 0.03){
          const g = ctx.createRadialGradient(lx, ly, 0, lx, ly, sz * 10);
          g.addColorStop(0,   light.color === '#D4A843' ? 'rgba(212,168,67,0.9)' : 'rgba(255,255,255,0.8)');
          g.addColorStop(0.3, light.color === '#D4A843' ? 'rgba(212,168,67,0.2)' : 'rgba(200,220,255,0.15)');
          g.addColorStop(1,   'transparent');
          ctx.fillStyle = g;
          ctx.beginPath();
          ctx.arc(lx, ly, sz * 10, 0, Math.PI * 2);
          ctx.fill();
          // Core bright dot
          ctx.beginPath();
          ctx.arc(lx, ly, sz * 0.7, 0, Math.PI * 2);
          ctx.fillStyle = light.color === '#D4A843' ? 'rgba(255,220,120,1)' : 'rgba(255,255,255,1)';
          ctx.fill();
        }
        light.pos += light.speed;
        if(light.pos > 1.05) light.pos = 0;
      });

      // ── Advance dashes ──
      dashOffset = (dashOffset + 0.012) % 1;

      requestAnimationFrame(draw);
    }

    draw();
  })();

  // Mobile menu
